// Simple RSS â†’ JSON Tide Proxy
// For Chester Tide Tables

const express = require("express");
const fetch = require("node-fetch");
const cors = require("cors");
const xml2js = require("xml2js");

const app = express();
app.use(cors());

const RSS_URL = "https://www.tidetimes.co.uk/chester-tide-times.rss";

app.get("/", (req, res) => {
  res.send("Tide Proxy Running");
});

app.get("/tides/chester", async (req, res) => {
  try {
    const xml = await fetch(RSS_URL).then(r => r.text());

    xml2js.parseString(xml, (err, data) => {
      if (err) return res.status(500).json({ error: "Parse failed" });

      const items = data.rss.channel[0].item;

      const parsed = items.map(it => {
        const title = it.title[0];
        const pubDate = it.pubDate[0];

        return {
          title,
          time: pubDate
        };
      });

      res.json({
        station: "Chester",
        tides: parsed
      });
    });
  } catch (e) {
    res.status(500).json({ error: e.toString() });
  }
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log("Tide Proxy Live on " + PORT));
